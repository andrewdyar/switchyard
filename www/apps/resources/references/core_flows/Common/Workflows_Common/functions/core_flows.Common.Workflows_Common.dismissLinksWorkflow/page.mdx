# dismissLinksWorkflow - Switchyard Core Workflows Reference

This documentation provides a reference to the `dismissLinksWorkflow`. It belongs to the `@switchyard/medusa/core-flows` package.

This workflow dismisses one or more links between records.

You can use this workflow within your customizations or your own custom workflows, allowing you to
dismiss links within your custom flows.

Learn more about links in [this documentation](https://docs.switchyard.com/learn/fundamentals/module-links/link).

import type {
  SwitchyardRequest,
  SwitchyardResponse,
} from "@switchyard/framework/http"
import { dismissLinksWorkflow } from "@switchyard/medusa/core-flows"

export async function POST(
  req: SwitchyardRequest,
  res: SwitchyardResponse
) {
  const { result } = await dismissLinksWorkflow(req.scope)
    .run({
      input: [{
        // import { Modules } from "@switchyard/framework/utils"
        [Modules.PRODUCT]: {
          product_id: "prod_123",
        },
        "helloModuleService": {
          my_custom_id: "mc_123",
        },
      }]
    })

  res.send(result)
}
